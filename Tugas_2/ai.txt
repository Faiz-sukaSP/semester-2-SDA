#include "stack.h"
#include <stdio.h>
#include <string.h>

int main()
{
    int pilihan;

    printf("============================================\n");
    printf("         PROGRAM STACK - STRUKTUR DATA      \n");
    printf("============================================\n");
    printf("  1. Cek Kurung Seimbang  (Stack Array)     \n");
    printf("  2. Infix ke Postfix     (Stack Linked List)\n");
    printf("  3. Evaluasi Postfix     (Array & LL)       \n");
    printf("  0. Keluar                                  \n");
    printf("============================================\n");

    do {
        printf("\nPilih menu [0-3]: ");
        scanf("%d", &pilihan);
        getchar(); // bersihkan newline

        switch (pilihan)
        {
        /*
         * ============================================
         *  POIN 1 — CEK KURUNG SEIMBANG (Stack Array)
         * ============================================
         */
        case 1:
        {
            char ekspresi[MAX];
            printf("\n--- CEK KURUNG SEIMBANG ---\n");
            printf("Masukkan ekspresi kurung: ");
            fgets(ekspresi, MAX, stdin);
            ekspresi[strcspn(ekspresi, "\n")] = '\0'; // hapus newline

            if (cek_kurung(ekspresi))
                printf("Hasil : Ekspresi \"%s\" SEIMBANG ✓\n", ekspresi);
            else
                printf("Hasil : Ekspresi \"%s\" TIDAK SEIMBANG ✗\n", ekspresi);
            break;
        }

        /*
         * ============================================
         *  POIN 2 — INFIX KE POSTFIX (Stack LinkedList)
         * ============================================
         */
        case 2:
        {
            char infix[MAX], postfix[MAX];
            printf("\n--- KONVERSI INFIX KE POSTFIX ---\n");
            printf("Masukkan ekspresi infix  : ");
            fgets(infix, MAX, stdin);
            infix[strcspn(infix, "\n")] = '\0';

            // Validasi kurung seimbang dulu
            if (!cek_kurung(infix))
            {
                printf("Error: Ekspresi infix memiliki kurung yang tidak seimbang!\n");
                break;
            }

            infix_to_postfix(infix, postfix);
            printf("Ekspresi infix   : %s\n", infix);
            printf("Ekspresi postfix : %s\n", postfix);
            break;
        }

        /*
         * ============================================
         *  POIN 3 — EVALUASI POSTFIX (Array & LL)
         * ============================================
         */
        case 3:
        {
            char infix[MAX], postfix[MAX];
            printf("\n--- EVALUASI EKSPRESI ---\n");
            printf("Masukkan ekspresi infix: ");
            fgets(infix, MAX, stdin);
            infix[strcspn(infix, "\n")] = '\0';

            if (!cek_kurung(infix))
            {
                printf("Error: Kurung tidak seimbang!\n");
                break;
            }

            infix_to_postfix(infix, postfix);
            printf("Ekspresi infix   : %s\n", infix);
            printf("Ekspresi postfix : %s\n", postfix);

            int hasil_arr = ev_post_arr(postfix);
            int hasil_ll  = ev_postLL(postfix);

            printf("Hasil (Array)    : %d\n", hasil_arr);
            printf("Hasil (LinkedList): %d\n", hasil_ll);
            break;
        }

        case 0:
            printf("\nTerima kasih! Program selesai.\n");
            break;

        default:
            printf("Pilihan tidak valid! Masukkan angka 0-3.\n");
        }

    } while (pilihan != 0);

    return 0;
}
