/* =========================
   STACK ARRAY
========================= */

void initArray(stack_array *s){
    s->top = -1;
}

int emptyArray(stack_array *s){
    return s->top == -1;
}

void pushArray(stack_array *s,int x){
    s->data[++s->top] = x;
}

int popArray(stack_array *s){
    return s->data[s->top--];
}

char popArrayChar(stack_array *s){
    return s->data[s->top--];
}

/* =========================
   STACK LINKED LIST
========================= */

void initStack(stack *ps){
    ps->top = NULL;
    ps->size = 0;
}

int emptyStack(stack *ps){
    return ps->top == NULL;
}

void push(stack *ps, int elemen){

    stack_nodeptr newNode = malloc(sizeof(stacknode));

    newNode->elemen = element;
    newNode->next = ps->top;

    ps->top = newNode;
    ps->size++;
}

int pop(stack *ps){

    stack_nodeptr temp = ps->top;
    int val = temp->elemen;

    ps->top = temp->next;

    free(temp);
    ps->size--;

    return val;
}

int peek(stack *ps){
    return ps->top->elemen;
}

/* =========================
   CEK KURUNG SEIMBANG
========================= */

int cekKurung(char expr[]){

    stack_array s;
    initArray(&s);

    for(int i=0; expr[i]; i++){

        char c = expr[i];

        if(c=='(' || c=='[' || c=='{')
            pushArray(&s,c);

        else if(c==')' || c==']' || c=='}'){

            if(emptyArray(&s))
                return 0;

            char top = popArrayChar(&s);

            if((c==')' && top!='(') ||
               (c==']' && top!='[') ||
               (c=='}' && top!='{'))
                return 0;
        }
    }

    return emptyArray(&s);
}

/* =========================
   PRIORITAS OPERATOR
========================= */

int precedence(char op){

    if(op=='+' || op=='-') return 1;
    if(op=='*' || op=='/') return 2;

    return 0;
}

/* =========================
   INFIX → POSTFIX
========================= */

void infixToPostfix(char infix[], char postfix[]){

    stack s;
    initStack(&s);

    int i=0,j=0;

    while(infix[i]){

        char c = infix[i];

        if(isalnum(c))
            postfix[j++] = c;

        else if(c=='(')
            push(&s,c);

        else if(c==')'){

            while(!emptyStack(&s) && peek(&s)!='(')
                postfix[j++] = pop(&s);

            pop(&s);
        }

        else{

            while(!emptyStack(&s) &&
                  precedence(peek(&s)) >= precedence(c))
                postfix[j++] = pop(&s);

            push(&s,c);
        }

        i++;
    }

    while(!emptyStack(&s))
        postfix[j++] = pop(&s);

    postfix[j]='\0';
}

/* =========================
   EVALUASI POSTFIX ARRAY
========================= */

int evaluatePostfixArray(char postfix[]){

    stack_array s;
    initArray(&s);

    for(int i=0; postfix[i]; i++){

        char c = postfix[i];

        if(isdigit(c))
            pushArray(&s, c-'0');

        else{

            int b = popArray(&s);
            int a = popArray(&s);

            switch(c){

                case '+': pushArray(&s,a+b); break;
                case '-': pushArray(&s,a-b); break;
                case '*': pushArray(&s,a*b); break;
                case '/': pushArray(&s,a/b); break;
            }
        }
    }

    return popArray(&s);
}
